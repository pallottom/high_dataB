def import_specimen(session, donor_id, row):
    # First, get or create the cell type (e.g., PBMC)
    cell_type = session.query(CellSpecimen).filter_by(name="PBMC").first()
    if not cell_type:
        cell_type = CellSpecimen(name="PBMC")
        session.add(cell_type)
        session.flush()

    # Get or create the human donor type
    human_donor_type = session.query(DonorType).filter_by(donor_category="human").first()
    if not human_donor_type:
        # Create the base Specimen first with explicit type
        specimen = Specimen(type='donor')
        session.add(specimen)
        session.flush()
        
        # Then create DonorType with the same ID
        human_donor_type = DonorType(
            id=specimen.id,  # Use the same ID
            donor_category="human",
            cell_type=cell_type
        )
        session.add(human_donor_type)
        session.flush()

    # Get or create the specific human donor
    human_donor = session.query(HumanDonor).filter_by(name=str(donor_id)).first()
    if not human_donor:
        human_donor = HumanDonor(
            name=str(donor_id),
            age=row.get("age"),
            gender=row.get("sex"),
        )
        session.add(human_donor)
        session.flush()
    
    return human_donor
